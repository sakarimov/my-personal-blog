Grid Search (Model Tuning)#

from sklearn.model_selection import GridSearchCV

model = SVR()
parameters = {
    'kernel': ['rbf'],
    'C': [1000, 10000, 100000],
    'gamma': [0.5, 0.05, 0.005]
}

grid_search = GridSearchCV(model, parameters)

grid_search.fit(X,y)
